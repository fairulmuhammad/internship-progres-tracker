// Firestore Security Rules for Internship Progress Tracker
// Copy these rules to Firebase Console > Firestore Database > Rules
// Updated to support both Google Sign-in and Anonymous users

rules_version = '2';
service cloud.firestore {
  match /databases/{database}/documents {
    // Allow authenticated users (both Google and Anonymous) to access their own data
    match /users/{userId} {
      allow read, write: if request.auth != null && request.auth.uid == userId;
      
      // Allow users to manage their own memos
      match /memos/{memoId} {
        allow read, write: if request.auth != null && request.auth.uid == userId;
      }
    }
    
    // Legacy support for your existing artifacts structure (if needed)
    match /artifacts/{appId}/users/{userId}/{document=**} {
      allow read, write: if request.auth != null && request.auth.uid == userId;
    }
    
    // Deny all other access
    match /{document=**} {
      allow read, write: if false;
    }
  }
}

// Key Changes:
// 1. Added support for users/{userId}/memos/{memoId} structure (our app's format)
// 2. Kept your existing artifacts/{appId}/users/{userId} structure for backward compatibility
// 3. Works with both Google Sign-in AND Anonymous authentication
// 4. request.auth != null covers both authenticated user types
